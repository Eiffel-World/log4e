	description: "Filter for log events to determine if they should be logged or not."
	project: "Project Goanna <http://sourceforge.net/projects/goanna>"

	date: "$Date$"

deferred class interface
	L4E_FILTER

feature -- Status report

	decide (event: L4E_EVENT): INTEGER
			-- Should 'event' be logged. Return one of Filter_accept,
			-- Filter_reject, or Filter_neutral.
		require
			event_exists: event /= void
		ensure
			valid_result: valid_filter_decision (Result)
	
feature -- Filter Constants

	Filter_accept: INTEGER is unique
			-- Event will be logged. Remaining filters will not
			-- be consulted.
			-- (from L4E_FILTER_CONSTANTS)

	Filter_neutral: INTEGER is unique
			-- Remaining filters will be consulted to determine
			-- if the event should be logged. If there are no
			-- remaining filters then the event will be logged.
			-- (from L4E_FILTER_CONSTANTS)

	Filter_reject: INTEGER is unique
			-- Event will not be logged.
			-- (from L4E_FILTER_CONSTANTS)

	valid_filter_decision (code: INTEGER): BOOLEAN
			-- Is 'code' a valid filter result code?
			-- (from L4E_FILTER_CONSTANTS)
	
invariant

		-- from ANY
	reflexive_equality: standard_is_equal (Current)
	reflexive_conformance: conforms_to (Current)

end -- class L4E_FILTER
